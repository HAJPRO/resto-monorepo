<script setup>
import { ref, reactive, onMounted, onUnmounted } from "vue";
import { useRouter } from "vue-router";
import { IonPage, IonContent, IonRefresher, IonRefresherContent } from '@ionic/vue';
import { Haptics, ImpactStyle } from '@capacitor/haptics';

const router = useRouter();
const credentials = reactive({ companyCode: "", username: "", password: "" });
const showPassword = ref(false);
const isLoading = ref(false);
const isDarkMode = ref(true);
const isMobile = ref(false);
const parallaxStyle = ref({});

const checkDevice = () => { isMobile.value = window.innerWidth < 1024; };

const toggleTheme = async () => {
  isDarkMode.value = !isDarkMode.value;
  localStorage.setItem('theme', isDarkMode.value ? 'dark' : 'light');
  document.documentElement.classList.toggle('dark', isDarkMode.value);
  await Haptics.impact({ style: ImpactStyle.Light });
};

const handleRefresh = async (event) => {
  await Haptics.impact({ style: ImpactStyle.Medium });
  Object.keys(credentials).forEach(k => credentials[k] = "");
  setTimeout(() => event.target.complete(), 1000);
};

const handleMouseMove = (e) => {
  if (isMobile.value) return;
  const x = (e.clientX / window.innerWidth - 0.5) * 20;
  const y = (e.clientY / window.innerHeight - 0.5) * 20;
  parallaxStyle.value = { transform: `rotateY(${x}deg) rotateX(${-y}deg)` };
};

const handleLogin = async () => {
  isLoading.value = true;
  await Haptics.impact({ style: ImpactStyle.Medium });
  
  setTimeout(async () => {
    isLoading.value = false;
    await router.replace({ name: 'home' });
  }, 1500);
};

const particleStyle = (n) => ({
  left: `${Math.random() * 100}%`,
  animationDuration: `${Math.random() * 10 + 5}s`,
  animationDelay: `${Math.random() * 5}s`,
  width: `${Math.random() * 3 + 1}px`,
  height: `${Math.random() * 3 + 1}px`,
});

onMounted(() => {
  checkDevice();
  window.addEventListener('resize', checkDevice);
  isDarkMode.value = localStorage.getItem('theme') !== 'light';
  document.documentElement.classList.toggle('dark', isDarkMode.value);
});

onUnmounted(() => window.removeEventListener('resize', checkDevice));
